    <script>
        // ==================== 🔥🔥🔥 把下面两行换成你自己的 🔥🔥🔥 ====================
        const BMOB_APP_ID = "7bd7ae53c0ba8b342172ffd7c9c80900";
        const BMOB_REST_API_KEY = "49200376a40c06b15a247c4ab50f1a3b";
        // =========================================================================

        const BMOB_API = `https://api2.bmobapp.com/1/classes/Rooms`;

        let currentRoomId = null;
        let pollingTimer = null;

        // DOM 元素
        const roomUI = document.getElementById('roomUI');
        const mainUI = document.getElementById('mainUI');
        const roomIdInput = document.getElementById('roomIdInput');
        const joinBtn = document.getElementById('joinRoomBtn');
        const leaveBtn = document.getElementById('leaveRoomBtn');
        const roomDisplay = document.getElementById('roomDisplay');
        const syncStatus = document.getElementById('syncStatus');
        const dishListEl = document.getElementById('dishList');
        const dishNameInput = document.getElementById('dishName');
        const dishImageInput = document.getElementById('dishImage');
        const addBtn = document.getElementById('addBtn');
        const randomBtn = document.getElementById('randomBtn');
        const resultBox = document.getElementById('resultBox');
        const resultName = document.getElementById('resultName');
        const resultImg = document.getElementById('resultImg');

        // 通用请求头
        function getHeaders() {
            return {
                "X-Bmob-Application-Id": BMOB_APP_ID,
                "X-Bmob-REST-API-Key": BMOB_REST_API_KEY,
                "Content-Type": "application/json"
            };
        }

        // ---- 房间操作 ----
        joinBtn.addEventListener('click', () => {
            const id = roomIdInput.value.trim();
            if (!id) { alert('请输入房间号'); return; }
            enterRoom(id);
        });
        roomIdInput.addEventListener('keydown', e => { if (e.key === 'Enter') joinBtn.click(); });

        leaveBtn.addEventListener('click', () => {
            clearInterval(pollingTimer);
            pollingTimer = null;
            currentRoomId = null;
            mainUI.style.display = 'none';
            roomUI.style.display = 'block';
            roomIdInput.value = '';
            resultBox.style.display = 'none';
        });

        async function enterRoom(roomId) {
            currentRoomId = roomId;
            roomUI.style.display = 'none';
            mainUI.style.display = 'block';
            roomDisplay.textContent = `房间：${roomId}`;

            // 检查是否已有该房间号的记录
            const res = await fetch(`${BMOB_API}?where={"roomId":"${roomId}"}`, { headers: getHeaders() });
            const data = await res.json();
            const results = data.results || [];

            // 如果没有，则新建一条
            if (results.length === 0) {
                await fetch(BMOB_API, {
                    method: 'POST',
                    headers: getHeaders(),
                    body: JSON.stringify({ roomId, dishes: "[]" })
                });
            }

            // 开始轮询同步（每2秒拉一次）
            fetchAndRender();
            pollingTimer = setInterval(fetchAndRender, 2000);
        }

        async function fetchAndRender() {
            if (!currentRoomId) return;
            try {
                const res = await fetch(`${BMOB_API}?where={"roomId":"${currentRoomId}"}`, { headers: getHeaders() });
                const data = await res.json();
                if (data.results && data.results.length > 0) {
                    const dishesStr = data.results[0].dishes || "[]";
                    const dishes = JSON.parse(dishesStr);
                    renderList(dishes);
                    syncStatus.textContent = '🟢 已连接';
                } else {
                    renderList([]);
                }
            } catch (e) {
                syncStatus.textContent = '🔴 连接失败，重试中...';
            }
        }

        // ---- 数据操作 ----
        async function addDish() {
            const name = dishNameInput.value.trim();
            if (!name) { alert('请输入菜名'); return; }
            const file = dishImageInput.files[0];
            let imageBase64 = null;
            if (file) {
                if (file.size > 2 * 1024 * 1024) { alert('图片不能超过2MB'); return; }
                imageBase64 = await fileToBase64(file);
            }
            const newDish = { name, image: imageBase64, addedBy: Date.now() };

            // 获取当前房间记录
            const res = await fetch(`${BMOB_API}?where={"roomId":"${currentRoomId}"}`, { headers: getHeaders() });
            const data = await res.json();
            if (!data.results || data.results.length === 0) { alert('房间丢失'); return; }
            const record = data.results[0];
            const dishes = JSON.parse(record.dishes || "[]");
            dishes.push(newDish);

            // 更新记录
            await fetch(`${BMOB_API}/${record.objectId}`, {
                method: 'PUT',
                headers: getHeaders(),
                body: JSON.stringify({ dishes: JSON.stringify(dishes) })
            });

            dishNameInput.value = '';
            dishImageInput.value = '';
            fetchAndRender(); // 立即刷新
        }

        async function deleteDish(index) {
            const res = await fetch(`${BMOB_API}?where={"roomId":"${currentRoomId}"}`, { headers: getHeaders() });
            const data = await res.json();
            if (!data.results || data.results.length === 0) return;
            const record = data.results[0];
            const dishes = JSON.parse(record.dishes || "[]");
            dishes.splice(index, 1);
            await fetch(`${BMOB_API}/${record.objectId}`, {
                method: 'PUT',
                headers: getHeaders(),
                body: JSON.stringify({ dishes: JSON.stringify(dishes) })
            });
            fetchAndRender();
        }

        function fileToBase64(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = e => resolve(e.target.result);
                reader.onerror = reject;
                reader.readAsDataURL(file);
            });
        }

        // ---- 渲染 ----
        function renderList(dishes) {
            if (!dishes || dishes.length === 0) {
                dishListEl.innerHTML = '<div class="empty-tip">还没有菜，快去添加吧～</div>';
                return;
            }
            dishListEl.innerHTML = dishes.map((dish, index) => `
                <div class="dish-card">
                    <button class="delete-btn" data-index="${index}">×</button>
                    ${dish.image 
                        ? `<img class="dish-img" src="${dish.image}" alt="${dish.name}">` 
                        : '<div class="dish-img" style="background:#f5f5f5; display:flex; align-items:center; justify-content:center; font-size:30px;">🍽️</div>'}
                    <div class="dish-name">${dish.name}</div>
                </div>
            `).join('');
            document.querySelectorAll('.delete-btn').forEach(btn => {
                btn.addEventListener('click', function(e) {
                    e.stopPropagation();
                    const idx = parseInt(this.dataset.index);
                    if (confirm('确定删除这道菜吗？')) deleteDish(idx);
                });
            });
        }

        // ---- 随机 ----
        async function randomPick() {
            const res = await fetch(`${BMOB_API}?where={"roomId":"${currentRoomId}"}`, { headers: getHeaders() });
            const data = await res.json();
            if (!data.results || data.results.length === 0) { alert('暂无数据'); return; }
            const dishes = JSON.parse(data.results[0].dishes || "[]");
            if (dishes.length === 0) { alert('还没有菜品'); return; }
            const picked = dishes[Math.floor(Math.random() * dishes.length)];
            resultName.textContent = picked.name;
            if (picked.image) {
                resultImg.src = picked.image;
                resultImg.style.display = 'block';
            } else {
                resultImg.style.display = 'none';
            }
            resultBox.style.display = 'block';
            resultBox.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        addBtn.addEventListener('click', addDish);
        randomBtn.addEventListener('click', randomPick);
        dishNameInput.addEventListener('keydown', e => { if (e.key === 'Enter') addDish(); });
    </script>
